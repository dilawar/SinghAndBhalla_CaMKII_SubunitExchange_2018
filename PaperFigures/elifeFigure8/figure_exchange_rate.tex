\begin{luacode}
    function sniffCode(filename)
        local f = io.open(filename, "r");
        local txt = f:read("*all");
        local tau = string.match(txt, 'tau%s+=%s*(%d+.%d+)');
        tex.write( string.format('%.1f', tau) );
    end
\end{luacode}
